cd /etc/yum.repos.d
#clickhouse.repo
[clickhouse-stable]
name=ClickHouse - Stable Repository
baseurl=http://repo.clickhouse.tech/rpm/stable/$basearch/

sudo -u clickhouse clickhouse-server --config-file=/etc/clickhouse-server/config.xml
